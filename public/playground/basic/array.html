var 아톰 = {

이름: '아톰,

개발: '깃허브',

공개: 2014,

색: '초록',

터미널: '없음'

}

var 비주얼_스튜디오 = {

이름: '비주얼 스튜디오',

개발: '마이크로소프트',

공개: 2015,

색: '파랑',

터미널: '있음'

}

console.log( 아톰 )

console.log( 비주얼_스튜디오 )

console.log( 아톰.이름 )

console.log( 비주얼_스튜디오.이름 )

'아톰/ 비주얼_스튜디오'라는 변수에 객체를 저장한 것
값이 수일 때는 문자값 보다 숫자값으로 쓰는 것이 더 적합하다.
객체는 값들을 그룹화할 때 쓰기 좋으며, 값에 대한 모든 것을 넣을 수 있다.

var 아톰 = {

이름: '아톰,

함수: function 편집기 () {

console.log( '바둑이방울' )
}

}

var 비주얼_스튜디오 = {

이름: '비주얼 스튜디오',

함수: function 편집기 () {

console.log( '잘도울린다' )
}

}

console.log( 아톰 )

console.log( 비주얼_스튜디오 )

console.log( 아톰.함수 )

console.log( 아톰[ '함수' ] )

console.log( 아톰.'함수'( ) )

속성은 대괄호나 .으로 접근한다.
객체 안에 있는 것들을 검색할 때 .을 쓴다.

변수는 대괄호 안에 쓸 수 있지만, 점 뒤에는 쓸 수 없다.

var 아톰 = {

이름: '아톰,

함수: function 편집기 () {

console.log( '바둑이방울' )
}

}

var 값 = '이름' ;
